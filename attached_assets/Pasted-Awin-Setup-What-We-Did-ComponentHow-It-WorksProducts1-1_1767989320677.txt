javascript// Fetch CJ promotions (similar to Awin)
async function fetchCJPromotions() {
  const response = await fetch('https://link-search.api.cj.com/v2/link-search', {
    headers: { 'Authorization': `Bearer ${CJ_API_TOKEN}` },
    params: {
      'website-id': CJ_WEBSITE_ID,
      'link-type': 'promotion',
      'promotion-type': 'coupon,sale'
    }
  });
  return response.json();
}
